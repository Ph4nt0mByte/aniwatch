### 1. Main Function – Get Streaming Sources with Fallbacks

```javascript
const API_BASE = "https://hianime-clone-kappa.vercel.app/api";

// Helper to fetch with timeout
const fetchWithTimeout = (url, timeout = 8000) => {
  return Promise.race([
    fetch(url),
    new Promise((_, reject) =>
      setTimeout(() => reject(new Error("Request timeout")), timeout)
    )
  ]);
};

/**
 * Get streaming sources for an episode
 * @param {string} animeId - e.g. "solo-leveling-17877"
 * @param {string|number} episodeId - The episode data id (important!)
 * @param {string} [server="hd-1"] 
 * @param {string} [type="sub"]   // "sub" or "dub"
 */
async function getStreamingSources(animeId, episodeId, server = "hd-1", type = "sub") {
  const params = new URLSearchParams({
    id: `${animeId}?ep=${episodeId}`,
    server: server,
    type: type
  });

  const url = `${API_BASE}/stream?${params.toString()}`;

  try {
    const res = await fetchWithTimeout(url);
    
    if (!res.ok) throw new Error(`HTTP ${res.status}`);

    const data = await res.json();

    if (data.success && data.results?.streamingLink?.length > 0) {
      return {
        success: true,
        source: data.results.streamingLink[0],   // Main video source
        allSources: data.results.streamingLink,
        servers: data.results.servers || [],
        rawData: data
      };
    } else {
      throw new Error("No streaming link found");
    }
  } catch (error) {
    console.warn(`Primary API failed for ${server} (${type}):`, error.message);
    return { success: false, error: error.message };
  }
}
```

### 2. Smart Player Function with Automatic Fallbacks

```javascript
async function loadEpisode(animeId, episodeId) {
  const serversToTry = ["hd-1", "hd-2", "backup"];
  const typesToTry = ["sub", "dub"];

  for (const type of typesToTry) {
    for (const server of serversToTry) {
      console.log(`Trying → Server: ${server} | Type: ${type}`);

      const result = await getStreamingSources(animeId, episodeId, server, type);

      if (result.success && result.source?.link?.file) {
        console.log("✅ Working source found!", result.source.link.file);
        
        // Play the video
        playVideo(result.source.link.file, result.source.tracks);
        return result;
      }
    }
  }

  console.error("❌ All sources failed");
  alert("Sorry, this episode is currently unavailable.");
}
```

### 3. Example: Playing the Video (with HLS support)

```javascript
function playVideo(videoUrl, tracks = []) {
  const videoPlayer = document.getElementById("anime-player"); // your <video> element

  if (Hls.isSupported() && videoUrl.includes(".m3u8")) {
    const hls = new Hls();
    hls.loadSource(videoUrl);
    hls.attachMedia(videoPlayer);
    
    // Add subtitles if available
    tracks.forEach(track => {
      if (track.kind === "captions") {
        const trackElement = document.createElement("track");
        trackElement.kind = "captions";
        trackElement.label = track.label;
        trackElement.src = track.file;
        trackElement.default = track.default;
        videoPlayer.appendChild(trackElement);
      }
    });
  } else {
    // Direct MP4 or fallback
    videoPlayer.src = videoUrl;
  }

  videoPlayer.play().catch(err => console.error(err));
}
```

### How to Use It

```javascript
// Example call
loadEpisode("solo-leveling-17877", "123456");   // animeId + episode data id
```

**Important Notes:**
- The most critical part is getting the correct **`episodeId`** (not just episode number 1, 2, 3). You usually get this from the `/api/info?id=...` endpoint under episodes list.
- You can also use `/api/servers/{animeId}?ep={episodeId}` to list available servers first.
