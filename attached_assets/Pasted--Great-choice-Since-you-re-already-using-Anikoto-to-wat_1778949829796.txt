### Anikoto API Base URL
**https://anikotoapi.site**

### Main Endpoints You Need

| Purpose                    | Endpoint                                      | Example |
|---------------------------|-----------------------------------------------|--------|
| Recent / Popular Anime    | `/recent-anime?page=1&per_page=20`           | https://anikotoapi.site/recent-anime?page=1&per_page=10 |
| Full Anime + Episodes     | `/series/{anime_id}`                          | https://anikotoapi.site/series/1 |
| Episode Streaming Links   | Inside the `/series/{id}` response            | `embed_url.sub` and `embed_url.dub` |

### JavaScript Implementation Example

```javascript
const ANIKOTO_BASE = "https://anikotoapi.site";

// 1. Get anime details + all episodes + embed links
async function getAnimeSeries(animeId) {
  try {
    const res = await fetch(`${ANIKOTO_BASE}/series/${animeId}`);
    const data = await res.json();

    if (data.ok && data.anime) {
      return {
        success: true,
        anime: data.anime,
        episodes: data.episodes || [],   // Each episode has embed_url.sub / embed_url.dub
      };
    } else {
      throw new Error("Failed to fetch anime");
    }
  } catch (error) {
    console.error("Anikoto API Error:", error);
    return { success: false, error: error.message };
  }
}

// 2. Example: Load a specific episode
async function loadEpisode(animeId, episodeNumber) {
  const result = await getAnimeSeries(animeId);

  if (!result.success) return;

  // Find the episode
  const episode = result.episodes.find(ep => ep.number === episodeNumber);

  if (episode?.embed_url?.sub) {
    const videoUrl = episode.embed_url.sub;   // or .dub
    console.log("✅ Streaming URL:", videoUrl);
    
    playVideo(videoUrl);   // Call your player function
  } else {
    console.log("No embed found for this episode");
  }
}

// 3. Simple video player function (iframe or <video>)
function playVideo(embedUrl) {
  const playerContainer = document.getElementById("player-container");
  
  // If it's an iframe embed link
  playerContainer.innerHTML = `
    <iframe 
      src="${embedUrl}" 
      width="100%" 
      height="500" 
      frameborder="0" 
      allowfullscreen>
    </iframe>`;
}
```

### How to Get the Anime ID
1. Call `/recent-anime` first to see examples.
2. Or search on anikoto.tv website → copy the ID from the URL (e.g. `solo-leveling-ilh08` or a number).
3. Then use that ID in `/series/{id}`.
